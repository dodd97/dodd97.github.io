  • 父类的变更会在不经意间影响子类

  • 设计原则: 识别应用中变化的方面,把它们和不变的方面分开

  • 设计原则: 针对接口编程,而不是针对实现编程

  • Duck行为整合

    • 行为接口类与实现

      // fly行为接口类
public interface IFlyBehavior
{
    // 抽象方法
    public abstract void fly();
}

// makeSound行为接口类
public interface IMakeSoundBehavior
{
    public abstract void makeSound();
}

    • 行为实现类

      public class FlyWithWings : IFlyBehavior
{
    public void fly()
    {
        Console.WriteLine("翅膀飞");
    }
}
public class FlyNoWay : IFlyBehavior
{
    public void fly()
    {
        Console.WriteLine("不能飞");
    }
}

public class Quack : IMakeSoundBehavior
{
    public void makeSound()
    {
        Console.WriteLine("嘎嘎叫");
    }
}
public class Squeak : IMakeSoundBehavior
{
    public void makeSound()
    {
        Console.WriteLine("吱吱叫");
    }
}
public class Mute : IMakeSoundBehavior
{
    public void makeSound()
    {
        Console.WriteLine("不能叫");
    }
}

    • 抽象父类

    // 抽象父类
public abstract class Duck
{
    // 行为接口对象
    public IFlyBehavior flyBehavior;
    public IMakeSoundBehavior makeSoundBehavior;

    // 通用抽象方法
    public abstract void display();
    
    public void swim()
    {
        Console.WriteLine("是鸭子就能游");
    }

    // 父类把方法委托给行为接口对象来调用
    public void performFly()
    {
        flyBehavior.fly();
    }
    public void performMakeSound()
    {
        makeSoundBehavior.makeSound();
    }
}
    • 子类
    public class MallardDuck : Duck
{
    public MallardDuck()
    {
        // 初始化时需要对行为类赋值
        flyBehavior = new FlyWithWings();
        makeSoundBehavior = new Quack();
    }

    // 实现父类的通用抽象方法
    public override void display()
    {
        Console.WriteLine("绿头鸭子");
    }
}
public class RubberDuck : Duck
{
    public RubberDuck()
    {
        flyBehavior = new FlyNoWay();
        makeSoundBehavior = new Squeak();
    }
    public override void display()
    {
        Console.WriteLine("橡皮鸭子");
    }
}
public class WoodenDuck : Duck
{
    public WoodenDuck()
    {
        // 初始化时需要对行为类赋值
        flyBehavior = new FlyNoWay();
        makeSoundBehavior = new Mute();
    }
    public override void display()
    {
        Console.WriteLine("木头鸭子");
    }
}
    • 对象新建与行为调用
    Duck mallardDuck = new MallardDuck();
mallardDuck.display();							// 绿头鸭子
mallardDuck.performFly();						// 翅膀飞
mallardDuck.performMakeSound();					// 嘎嘎叫

Duck rubberDuck = new RubberDuck();
rubberDuck.display();							// 橡皮鸭子
rubberDuck.performFly();						// 不能飞
rubberDuck.performMakeSound();					// 吱吱叫

Duck woodenDuck = new WoodenDuck();
woodenDuck.display();							// 木头鸭子
woodenDuck.performFly();						// 不能飞
woodenDuck.performMakeSound();					// 不能叫

  • 这种实现方法会让其他类型的对象可以复用fly和makeSound方法,因为行为不再被限制在类里面.也可以直接新增行为,而不修改任何已有的行为类

  • 如果给一只木头鸭增加了一个推进器,让它拥有了飞行的能力.

    • 给Duck新增换flyBehavior的方法
    public abstract class Duck
{
    public void setFlyBehavior(IFlyBehavior _flyBehavior)
    {
        flyBehavior = _flyBehavior;
    }
    public void setMakeSoundBehavior(IMakeSoundBehavior _makeSoundBehavior)
    {
        makeSoundBehavior = _makeSoundBehavior;
    }
}
    • 新增行为类
    public class FlyWithThruster : IFlyBehavior
{
    public void fly()
    {
        Console.WriteLine("用推进器飞");
    }
}
    • 赋予一只木头鸭飞的能力
    Duck woodenDuck = new WoodenDuck();
woodenDuck.performFly();					// 不能飞
IFlyBehavior flyBehavior = new FlyWithThruster();
woodenDuck.setFlyBehavior(flyBehavior);
woodenDuck.performFly();					// 用推进器飞

  • 设计原则: 优先使用组合而不是继承

    • HAS-A 比 IS-A 好

  • 以上,是 策略模式

    • 策略模式定义了一个算法族,分别封装起来,使得他们之间可以相互变换,策略让算法的变化独立于使用它的客户
