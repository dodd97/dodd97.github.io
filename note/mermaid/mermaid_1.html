  • 属性与函数

    // 常用修饰符
// +:Public; -:Private; #:Protected; ~:Package/Internal
// 后置: *:Abstract; $:Static

// 定义时追加
Class Car {
	+String Driver$
    +Run()*
}

// 单独追加
Car: ~String AI

// 方法的参数与返回类型
Car: -Whistle(int times) bool

// 泛型
Car: #List~Wheel~ wheels
Car: +Aboard(List~Person~ people) int

M ..> N			// ..> : 虚线箭头
